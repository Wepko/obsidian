```
# 1. Посмотреть товары с фильтром
GET /api/v1/products?category=Engine&search=Filter

# 2. Создать заказ
POST /api/v1/orders
{
    "customer_id": 1,
    "items": [
        {"product_id": 1, "quantity": 2},
        {"product_id": 2, "quantity": 1}
    ]
}

# 3. Посмотреть заказы
GET /api/v1/orders?status=new

# 4. Подтвердить заказ (менеджер)
PATCH /api/v1/orders/1/status
{
    "status": "confirmed"
}

# 5. Автоматически запускается ExportOrderJob (через очередь)
```
