```python
# Дано:
distance_km = 10  # дистанция в километрах
time_minutes = 42  # минуты
time_seconds = 42  # секунды

# 1. Переводим километры в мили (1 км ≈ 0.621371 мили)
distance_miles = distance_km * 0.621371

# 2. Переводим время в часы и минуты
total_minutes = time_minutes + time_seconds / 60  # всего минут (42.7)
total_hours = total_minutes / 60  # всего часов (0.7117)

# 3. Рассчитываем средний темп (время на милю)
pace_minutes = total_minutes / distance_miles  # минут на милю (6.87)
minutes = int(pace_minutes)  # целая часть (6)
seconds = (pace_minutes - minutes) * 60  # остаток в секундах (~52)

# 4. Рассчитываем среднюю скорость (миль в час)
speed_mph = distance_miles / total_hours

# Вывод результатов
print(f"Средний темп: {minutes} мин {round(seconds)} сек на милю")
print(f"Средняя скорость: {round(speed_mph, 2)} миль/час")
```
