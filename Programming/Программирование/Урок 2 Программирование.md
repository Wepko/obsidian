### **–ö–æ–Ω–≤–µ—Ä—Ç–µ—Ä —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—ã (–¶–µ–ª—å—Å–∏–π ‚Üî –§–∞—Ä–µ–Ω–≥–µ–π—Ç)**  

#### **üîπ –ù–∞ C**  
```c

    int choice;

    printf("1. –¶–µ–ª—å—Å–∏–π ‚Üí –§–∞—Ä–µ–Ω–≥–µ–π—Ç\n");
    printf("2. –§–∞—Ä–µ–Ω–≥–µ–π—Ç ‚Üí –¶–µ–ª—å—Å–∏–π\n");
    printf("–í—ã–±–µ—Ä–∏—Ç–µ –æ–ø—Ü–∏—é (1 –∏–ª–∏ 2): ");
    scanf("%d", &choice);

    if (choice == 1) {
        printf("–í–≤–µ–¥–∏—Ç–µ —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—É –≤ –¶–µ–ª—å—Å–∏—è—Ö: ");
        scanf("%f", &celsius);
        fahrenheit = (celsius * 9 / 5) + 32;
        printf("%.2f¬∞C = %.2f¬∞F\n", celsius, fahrenheit);
    } else if (choice == 2) {
        printf("–í–≤–µ–¥–∏—Ç–µ —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—É –≤ –§–∞—Ä–µ–Ω–≥–µ–π—Ç–∞—Ö: ");
        scanf("%f", &fahrenheit);
        celsius = (fahrenheit - 32) * 5 / 9;
        printf("%.2f¬∞F = %.2f¬∞C\n", fahrenheit, celsius);
    } else {
        printf("–û—à–∏–±–∫–∞: –Ω–µ–≤–µ—Ä–Ω—ã–π –≤—ã–±–æ—Ä.\n");
    }

    return 0;
}
```
